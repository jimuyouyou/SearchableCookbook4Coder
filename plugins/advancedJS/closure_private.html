function Ninja() {
		var slices = 0;
		this.getSlices = function(){
			return slices;
		}; 

		this.slice = function(){
			slices++;
		};
	}

	var ninja = new Ninja();
	ninja.slice();
	
	assert(ninja.getSlices() == 1, "We're able to access the internal slice data." );
	assert(ninja.slices === undefined, "And the private data is inaccessible to us." );

	#1 Defines constructor
	#2 Declares “private” variable
	#3 Creates an accessor method
	#4 Declares increment method
	#5 Constructs the object
	#6 Calls the increment method
	#7 Tests the private-ness
