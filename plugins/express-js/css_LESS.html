.borderIt (@rad:0, @size:1px, @type: solid, @col: #000) {
	border: @size @type @col;
}

.borderIt (@rad:0) when (@rad > 0) {
	-webkit-border-radius: @rad;
	-moz-border-radius: @rad;
	border-radius: @rad;
}

	.borderIt(20px, 2px);
	.pad();
	th { .pad(); }
	td { .pad();

	}
