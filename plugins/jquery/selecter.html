$('pre').show; $('p').hide();
		$("input[id^='code']");//id属性以code开始的所有input标签
		$("input[id$='code']");//id属性以code结束的所有input标签
		$("input[id*='code']");//id属性包含code的所有input标签
		$(':button') ，所有的 button 元素都会被选择
		$(':checkbox') ，所有的checkbox 元素都会被选择。
		$(':checked') ，可以看做是 Checkbox 选择器的补充，用于选择所有已经被勾选的 Checkbox 对象。
		$("tbody tr:even"); //选择索引为偶数的所有tr标签
		$("tbody tr:odd"); //选择索引为奇数的所有tr标签
		$('tr:odd')为取得所有奇数行。
		$('tr:even')为取得所有偶数行。
		$('td:contains("hahaha")')为取得所有单元格包含hahaha的元素。
		$("#code input:not([id^='code'])");//id为code标签内不包含id以code开始的所有input标签
		$('#title1 > li')为取得ID为title1(#title)的子元素(>)中所有的列表项(li)。
		$('#title1 li:not(.class1)')为取得ID为title的后代元素中没有(not)class1类的所有列表项。
		$(".main > a");
		$('a[@title]')为取得所有带title属性的链接。
		$('a[href^="mailto:"]')为取得所有电子邮件的链接。
		$('a[href$=".rar"]')为取得所有指向RAR文件的链接。
		$('a[href*="tao028.com"]')为取得所有包含tao028.com的链接。
		$('div[ol]')为取得所以包含一个ol属性的div元素。
		$('div:eq(1)')为取得div元素集合中第2个项，注意为单个对象，其length为1。
		$('div:nth-child(1)')为取得div元素集合中的所有第一个div，注意其为对象集合，稍后再详述。
		$('[attribute^=value]') ，用于选择属性的值以某个字符串开头的元素
		$('[rel*=no]')，表示所有带有 rel 属性，并且 rel 的值里包含子字符串 “no” 的元素（如 rel=”nofollow”，rel=”yesorno” 等等）都将会被选择
		$('[rel$=no]') 将会选择 rel 属性的值以 “no” 结尾的元素。
		$('[rel=nofollow]')，则只选择 rel=”nofollow” 的元素，差一点都不行！
		$('[rel!=nofollow]')，所有 rel=”nofollow” 的元素都不会被选择。
		$(Obj).next("div");//获取$Obj标签的后面紧邻的一个div，nextAll获取所有
		$('parent > child')，只选择 parent 的子元素（下一级元素），不对子元素以下的元素进行选择
		$(Obj).children("input").length; //获得$Obj下一级节点的input个数
